public function filtrarPorMes(Request $request)
{
    // Validar los meses de entrada
    $request->validate([
        'mesInicio' => 'required|integer|min:1|max:12',
        'mesFin' => 'required|integer|min:1|max:12',
    ]);

    // Obtener los meses y el año del formulario
    $mesInicio = $request->input('mesInicio');
    $mesFin = $request->input('mesFin');
    $year = $request->input('year') ?? date('Y'); // Usar el año actual si no se proporciona

    // Obtener los registros filtrados por mes y año
    $registros = RegistroAulas::where('year', $year)
        ->whereBetween('mes', [$mesInicio, $mesFin])
        ->get();

    // Agrupar los registros por aula y luego por mes
    $datosAgrupados = $registros->groupBy('aula')->map(function ($grupoPorAula) {
        return $grupoPorAula->groupBy('mes')->map(function ($grupoPorMes) {
            return $grupoPorMes->sum('alumnos');
        });
    });

    // Preparar las etiquetas y los datos para la gráfica
    $labels = [];
    $datasets = [];

    // Crear un dataset por cada aula, con las sumas de alumnos por mes
    foreach ($datosAgrupados as $aula => $meses) {
        $dataset = [
            'label' => $aula,
            'data' => [],
        ];

        // Llenar los datos del dataset para cada mes dentro del rango
        for ($mes = $mesInicio; $mes <= $mesFin; $mes++) {
            $dataset['data'][] = $meses->get($mes, 0); // Usar 0 si no hay datos para ese mes

    // Retornar la vista con los datos agrupados

{
    // Obtener los datos agrupados por mes y aula
    $datos = DB::table('registrosaulas')
        ->select('aula', 'mes', DB::raw('SUM(alumnos) AS total_alumnos'))
        ->groupBy('aula', 'mes')
        ->orderBy('aula')
        ->orderBy(DB::raw("FIELD(mes, 'Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio', 'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre')"))
        ->get();

    // Agrupar los datos por aula y luego por mes
    $datosAgrupados = $datos->groupBy('aula')->map(function ($grupoPorAula) {
        return $grupoPorAula->groupBy('mes')->mapWithKeys(function ($grupoPorMes) {
            return [$grupoPorMes->first()->mes => $grupoPorMes->sum('total_alumnos')];
        });
    });

    // Obtener la lista de meses y aulas
    $meses = $datos->pluck('mes')->unique()->sort();
    $aulas = $datos->pluck('aula')->unique()->sort();

    // Preparar los datos para la tabla
    $datosTabla = $aulas->mapWithKeys(function ($aula) use ($datosAgrupados, $meses) {
        $sumaPorAula = $meses->mapWithKeys(function ($mes) use ($aula, $datosAgrupados) {
            return [$mes => $datosAgrupados->get($aula, collect())->get($mes, 0)];
        });

        $sumaPorAula['Total'] = $sumaPorAula->sum();

        return [$aula => $sumaPorAula];
    });

    // Preparar los datos totales por mes
    $totalPorMes = $meses->mapWithKeys(function ($mes) use ($datosAgrupados) {
        return [$mes => $datosAgrupados->flatten()->sum($mes)];
    });

    $totalPorMes['Total'] = $totalPorMes->sum();

    return view('inform.alumnos', [
        'datosTabla' => $datosTabla,
